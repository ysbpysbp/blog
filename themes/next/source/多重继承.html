function multipleInheritance() {
    let parentClasses = [...arguments];

    function TargetClass() {
      for (var i = 0; i < parentClasses.length; i++) {
        parentClasses[i].call(this)
      }
    }

    // inherit one class
    TargetClass.prototype = Object.create(parentClasses[0].prototype);

    // mixin another
    for (var j = 0; j < parentClasses.length; j++) {
      Object.assign(TargetClass.prototype, parentClasses[j].prototype);
    }

    // re-assign constructor
    TargetClass.prototype.constructor = TargetClass;

    return TargetClass
  }

  function Animal() {
    this.type = '动物';
  }

  Animal.prototype = {
    run: function() {
      console.log('animal run')
    }
  }

  function Pet() {
    this.prop = '受主人宠爱';
  }

  Pet.prototype = {
    act: function() {
      console.log('撒娇')
    }
  }

  var AnimalPet = multipleInheritance(Animal, Pet);

  var animalPet = new AnimalPet();

  console.log('animalPet.type ', animalPet.type)
  console.log('animalPet.prop ', animalPet.prop)
  animalPet.run();
  animalPet.act();
